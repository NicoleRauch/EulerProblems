Problem 2


fiblist :: [Int] -> [Int]
fiblist [] = fiblist (1:1:[])
fiblist [x] = fiblist (1:1:[])
fiblist (x:y:xs)
 | x + y < 4000000 = fiblist ((x+y):x:y:xs)
 | otherwise = (x:y:xs)

problem2 = sum $ filter (\ v -> v `mod` 2 == 0) $ fiblist []
